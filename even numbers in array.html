        var arr = new Array();
        arr[0] = 1;
        arr[1] = 22;
        arr[3] = 33;
        arr[4] = 12;
        arr[5] = 14;
        
        for (let i = 0; i < arr.length; i++) {
            if (arr[i] !== undefined && arr[i] % 2 == 0) {
                document.write(arr[i] + ", ");
            }
        }
